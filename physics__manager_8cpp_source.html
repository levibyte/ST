33 void physics_manager::process_horizontal() {
34  for(auto& i : *entities) {
35  //handle horizontal velocity
36  if (i.is_affected_by_physics()) {
37  if (i.get_velocity_x() > 0) {
38  for (int j = 0; j < i.get_velocity_x(); j++) {
39  if (entity_set_x(i.get_x() + 1, i.get_ID(), entities) == 0) {
40  break;

42  }
43  for (int j = 0; j < friction && i.get_velocity_x() > 0; j++) {
44  (i.set_velocity_x(i.get_velocity_x() - 1));

63 void physics_manager::process_vertical() {
64  for(auto& i : *entities) {
65  if (i.is_affected_by_physics()) {
66  //handle vertical velocity
67  const int objectVelocity = i.get_velocity_y() + gravity;
68  if (objectVelocity < 0) {
69  for (int j = 0; j > objectVelocity; j--) {
70  if (entity_set_y(i.get_y() - 1, i.get_ID(), entities) == 0) {
71  break;
72  }
73  }
74  } else if (objectVelocity > 0) {
75  for (int j = 0; j < objectVelocity; j++) {
76  if (i.get_y() + i.get_col_y_offset() < level_floor) {
77  if (entity_set_y(i.get_y() + 1, i.get_ID(), entities) == 0) {
78  break;
79  }
80  }
81  }
82  }
83  //decrease velocity of objects (apply gravity really)
84  int realVelocity = objectVelocity - gravity;
85  if (realVelocity < 0) {
86  i.set_velocity_y(realVelocity + 2);
87  }
88  }
89  }

96 void physics_manager::handle_messages(){
97  message* temp = msg_sub.get_next_message();
98  while(temp != nullptr){
99  if(temp->msg_name == SET_GRAVITY){
100  gravity = *static_cast<int8_t*>(temp->get_data());
101  }else if(temp->msg_name == SET_FLOOR){
102  level_floor = *static_cast<int32_t*>(temp->get_data());
103  }else if(temp->msg_name == SET_FRICTION){
104  friction = *static_cast<int8_t*>(temp->get_data());
105  }else if(temp->msg_name == PAUSE_PHYSICS){
106  physics_paused = true;
107  }else if(temp->msg_name == UNPAUSE_PHYSICS){
108  physics_paused = false;
109  }
110  destroy_msg(temp);
111  temp = msg_sub.get_next_message();
112  }
113 }
