25 void physics_manager::process_horizontal() {
26  for(auto& i : *entities) {
27  //handle horizontal velocity
28  if (i.is_affected_by_physics()) {
29  if (i.get_velocity_x() > 0) {
30  for (int j = 0; j < i.get_velocity_x(); j++) {
31  if (entity_set_x(i.get_x() + 1, i.get_ID(), entities) == 0) {
32  break;
33  }
34  }
35  for (int j = 0; j < friction && i.get_velocity_x() > 0; j++) {
36  (i.set_velocity_x(i.get_velocity_x() - 1));
37  }
38  } else if (i.get_velocity_x() < 0) {
39  for (int j = 0; j > i.get_velocity_x(); j--) {
40  if (entity_set_x(i.get_x() - 1, i.get_ID(), entities) == 0) {
41  break;
42  }
43  }
44  for (int j = 0; j < friction && i.get_velocity_x() < 0; j++) {
45  (i.set_velocity_x(i.get_velocity_x() + 1));
46  }
47  }
48  }
49  }
50 }

52 void physics_manager::process_vertical() {
53  for(auto& i : *entities) {
54  if (i.is_affected_by_physics()) {
55  //handle vertical velocity
56  const int objectVelocity = i.get_velocity_y() + gravity;
57  if (objectVelocity < 0) {
58  for (int j = 0; j > objectVelocity; j--) {
59  if (entity_set_y(i.get_y() - 1, i.get_ID(), entities) == 0) {
60  break;
61  }
62  }
63  } else if (objectVelocity > 0) {
64  for (int j = 0; j < objectVelocity; j++) {
65  if (i.get_y() + i.get_col_y_offset() < level_floor) {
66  if (entity_set_y(i.get_y() + 1, i.get_ID(), entities) == 0) {
67  break;
68  }
69  }
70  }
71  }
72  //decrease velocity of objects (apply gravity really)
73  int realVelocity = objectVelocity - gravity;
74  if (realVelocity < 0) {
75  i.set_velocity_y(realVelocity + 2);
76  }
77  }
78  }
79 }

81 void physics_manager::handle_messages(){
82  message* temp = msg_sub.get_next_message();
83  while(temp != nullptr){
84  if(temp->msg_name == SET_GRAVITY){
85  gravity = *(int*)temp->get_data();
86  }else if(temp->msg_name == SET_FLOOR){
87  level_floor = *(int*)temp->get_data();
88  }else if(temp->msg_name == SET_FRICTION){
89  friction = *(int*)temp->get_data();
90  }else if(temp->msg_name == PAUSE_PHYSICS){
91  physics_paused = true;
92  }else if(temp->msg_name == UNPAUSE_PHYSICS){
93  physics_paused = false;
94  }
95  destroy_msg(temp);
96  temp = msg_sub.get_next_message();
97  }
98 }

124 int physics_manager::check_collision(uint64_t ID, std::vector<ST::entity>* entities){
125  for(unsigned int i = 0; i < entities->size(); i++){
126  ST::entity* temp = &entities->at(i);
127  if(temp->is_affected_by_physics()){
128  if(ID == i){
129  continue;
130  }
131  if(temp->collides(entities->at(ID))){
132  return 0;
133  }
134  }
135  }
