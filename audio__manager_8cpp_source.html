61 void audio_manager::handle_messages(){
62  message* temp = msg_sub.get_next_message();
63  while(temp != nullptr){
64  if(temp->msg_name == PLAY_SOUND){
65  auto data = static_cast<std::tuple<size_t, int, int>*>(temp->get_data());
66  size_t name = std::get<0> (*data);
67  int volume = std::get<1> (*data);
68  int loops = std::get<2> (*data);
69  play_sound(name, volume, loops);
70  }
71  else if(temp->msg_name == PLAY_MUSIC){
72  auto data = (std::tuple<size_t, int, int>*)temp->get_data();
73  size_t name = std::get<0> (*data);
74  int volume = std::get<1> (*data);
75  int loops = std::get<2> (*data);
76  play_music(name, volume, loops);
77  }
78  else if(temp->msg_name == STOP_MUSIC){
79  stop_music();
80  log(SUCCESS, "Music stopped");
81  }
82  else if(temp->msg_name == STOP_ALL_SOUNDS){
83  stop_channels();
84  log(SUCCESS, "Sounds stopped");
85  }
86  else if(temp->msg_name == TOGGLE_AUDIO){
87  if(volume > 0){
88  log(SUCCESS, "Audio OFF");
89  mute();
90  set_volume(0);
91  }
92  else{
93  set_volume(MIX_MAX_VOLUME);
94  unmute();
95  log(SUCCESS, "Audio ON");
96  }
97  gMessage_bus->send_msg(make_msg(VOLUME_LEVEL, make_data<int>(volume)));
98  }
99  else if(temp->msg_name == ASSETS){
100  auto temp_ptr = (ST::assets**)temp->get_data();
101  assets_ptr = *temp_ptr;
102  }else if(temp->msg_name == SET_VOLUME){
103  auto arg = static_cast<uint8_t*>(temp->get_data());
104  set_volume(*arg);
105  gMessage_bus->send_msg(make_msg(VOLUME_LEVEL, make_data<int>(volume)));
106  }
107  destroy_msg(temp);
108  temp = msg_sub.get_next_message();
109  }
110 }
