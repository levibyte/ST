155 inline void audio_manager::play_sound(size_t arg, uint8_t volume, int8_t loops) const{
156  auto data = chunks_ptr->find(arg);
157  if(data != chunks_ptr->end()){
158  if(!muted){
159  Mix_VolumeChunk(data->second, static_cast<int>(static_cast<float >(volume) / chunk_playback_volume_ratio));
160  }
161  if(Mix_PlayChannel( -1, data->second, loops ) == -1){
162  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("Mix_PlayChannel Error " + std::string(Mix_GetError()))));
163  }
164  }
165 }

173 inline void audio_manager::play_music(size_t arg, uint8_t volume, int8_t loops) const{
174  auto data = music_ptr->find(arg);
175  if(data != music_ptr->end() ){
176  if(!muted) {
177  Mix_VolumeMusic(static_cast<int>(static_cast<float>(volume) / music_playback_volume_ratio));
178  }
179  if(Mix_PlayMusic(data->second, loops) == -1){
180  gMessage_bus->send_msg(make_msg(LOG_ERROR, make_data<std::string>("Mix_PlayMusic Error " + std::string(Mix_GetError()))));
181  }
182  if(!muted) {
183  Mix_VolumeMusic(this->music_volume);
184  }
185  }
