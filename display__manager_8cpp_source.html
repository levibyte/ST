76 void display_manager::handle_messages(){
77  message* temp = msg_sub.get_next_message();
78  while(temp != nullptr){
79  if(temp->msg_name == SET_FULLSCREEN){
80  auto arg = static_cast<bool*>(temp->get_data());
81  set_fullscreen(*arg);
82  gMessage_bus->send_msg(make_msg(FULLSCREEN_STATUS, make_data<bool>(*arg)));
83  }
84  else if(temp->msg_name == SET_WINDOW_BRIGHTNESS){
85  auto arg = static_cast<float*>(temp->get_data());
86  set_brightness(*arg);
87  log(SUCCESS, "Brightness set to: " + std::to_string(*arg));
88  }
89  destroy_msg(temp);
90  temp = msg_sub.get_next_message();
91  }
92 }
