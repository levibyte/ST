77 void display_manager::handle_messages(){
78  message* temp = msg_sub.get_next_message();
79  while(temp != nullptr){
80  if(temp->msg_name == SET_FULLSCREEN){
81  auto arg = static_cast<bool*>(temp->get_data());
82  set_fullscreen(*arg);
83  gMessage_bus->send_msg(make_msg(FULLSCREEN_STATUS, make_data<bool>(*arg)));
84  }
85  else if(temp->msg_name == SET_WINDOW_BRIGHTNESS){
86  auto arg = static_cast<float*>(temp->get_data());
87  set_brightness(*arg);
88  log(SUCCESS, "Brightness set to: " + std::to_string(*arg));
89  }
90  destroy_msg(temp);
91  temp = msg_sub.get_next_message();
92  }
93 }
