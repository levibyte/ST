75 void display_manager::handle_messages(){
76  message* temp = msg_sub.get_next_message();
77  while(temp != nullptr){
78  if(temp->msg_name == SET_FULLSCREEN){
79  auto arg = static_cast<bool*>(temp->get_data());
80  set_fullscreen(*arg);
81  gMessage_bus->send_msg(make_msg(FULLSCREEN_STATUS, make_data<bool>(*arg)));
82  }
83  else if(temp->msg_name == SET_WINDOW_BRIGHTNESS){
84  auto arg = static_cast<float*>(temp->get_data());
85  set_brightness(*arg);
86  log(SUCCESS, "Brightness set to: " + std::to_string(*arg));
87  }
88  else if(temp->msg_name == SET_WINDOW_RESOLUTION){
89  auto res = *static_cast<std::tuple<int16_t, int16_t>*>(temp->get_data());
90  gMessage_bus->send_msg(make_msg(REAL_SCREEN_COORDINATES, make_data(res)));
91  SDL_SetWindowSize(window, std::get<0>(res), std::get<1>(res));
92  SDL_SetWindowPosition(window, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED);
93  width = std::get<0>(res);
94  height = std::get<1>(res);
95  }
96  destroy_msg(temp);
97  temp = msg_sub.get_next_message();
98  }
99 }
