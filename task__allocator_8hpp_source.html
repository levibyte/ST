35 inline ST::task* task_allocator::allocate_task(void (*function)(void *), void *arg, SDL_semaphore *dependency){
36  uint16_t i = 0;
37  SDL_LockMutex(access_mutex);
38  //find the next free spot in memory
39  while(allocated[pointer] && i < memory_size){
40  pointer++;
41  i++;
42  if(pointer > memory_size-1){
43  pointer = 0;
44  }
45  }
46  if(i == memory_size){
47  fprintf(stderr, "You have started more than %d tasks, exiting!", TASK_ALLOCATOR_CAPACITY);
48  exit(1);
49  }
50  allocated[pointer] = true;
51  auto temp = new (memory+pointer) ST::task(pointer, function, arg, dependency);
52  SDL_UnlockMutex(access_mutex);
53  return temp;
54 }
