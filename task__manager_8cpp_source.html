19 int task_manager::task_thread(void* arg){
20  auto self = static_cast<task_manager*>(arg);
21  ST::task* work;
22  while(SDL_AtomicGet(&self->run_threads) == 1){
23  SDL_SemWait(self->work_sem);
24  if(self->task_queue.try_dequeue(work)){ //get a function pointer and data

36 void task_manager::do_work(ST::task* work) {
37  if(work->dependency != nullptr){ //wait for dependency to finish
38  SDL_SemWait(work->dependency);
39  SDL_DestroySemaphore(work->dependency);
40  }
41  work->task_func(work->data); // call it
42  if(work->has_lock) {
43  SDL_SemPost(work->lock); //increment the semaphore
44  }
45  destroy_task(work);
46 }

92 task_manager::~task_manager(){

117 task_id task_manager::start_task(ST::task* arg){
