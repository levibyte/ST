43 void font_cache::cache_string(std::string STR, SDL_Texture* TEXTURE, std::string FONT, int SIZE){
44  font_cache_tuple temp = std::make_tuple(STR, FONT, SIZE);
45  cache.push_front(std::make_pair(temp, TEXTURE));
46  hash[temp] = cache.begin();
47  entries++;
48  if(entries > cache_size){
