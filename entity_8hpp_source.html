337 inline void ST::entity::set_collision_box(int16_t offsetX, int16_t offsetY, int16_t X, int16_t Y){
338  if(X != 0 && Y != 0){
339  col_x = X - 1;
340  col_y = -Y + 1;
341  offset_x = offsetX;
342  offset_y = offsetY;
343  }else{
344  col_x = 0;
345  col_y = 0;
346  set_affected_by_physics(false);
347  }
348  update_collision_box();
349 }

354 inline void ST::entity::update_collision_box(){

356  collision.right = x + col_x + offset_x;
357  collision.top = y + col_y + offset_y;
358  collision.bottom = y + offset_y;
