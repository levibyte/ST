341 inline void ST::entity::set_collision_box(int16_t offsetX, int16_t offsetY, int16_t X, int16_t Y){
342  if(X != 0 && Y != 0){
343  col_x = X - 1;
344  col_y = -Y + 1;
345  offset_x = offsetX;
346  offset_y = offsetY;
347  }else{
348  col_x = 0;
349  col_y = 0;
350  set_affected_by_physics(false);
351  }
352  update_collision_box();
353 }

358 inline void ST::entity::update_collision_box(){
359  collision.left = x + offset_x;
360  collision.right = x + col_x + offset_x;
361  collision.top = y + col_y + offset_y;
362  collision.bottom = y + offset_y;

386 inline bool ST::entity::collides(entity other) const{
387  return !((collision.bottom < other.collision.top) || (collision.top > other.collision.bottom)
388  || (collision.left > other.collision.right) || (collision.right < other.collision.left));
