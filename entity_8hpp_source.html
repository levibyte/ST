338 inline void ST::entity::set_collision_box(int16_t offsetX, int16_t offsetY, int16_t X, int16_t Y){
339  if(X != 0 && Y != 0){
340  col_x = X - 1;
341  col_y = -Y + 1;
342  offset_x = offsetX;
343  offset_y = offsetY;
344  }else{
345  col_x = 0;
346  col_y = 0;
347  set_affected_by_physics(false);
348  }
349  update_collision_box();
350 }
