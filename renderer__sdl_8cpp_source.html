94 void renderer_sdl::draw_text_lru_cached(std::string arg, std::string arg2, int x, int y, SDL_Color color_font, int size){
95  std::string font_and_size = arg+std::to_string(size);
96  TTF_Font* font = fonts[font_and_size];
97  if(font != nullptr){
98  int texW, texH;
99  SDL_Texture* cached_texture = gFont_cache.get_cached_string(arg2, arg, size);
100  if(cached_texture != nullptr){ //if the given string (with same size and font) is already cached, get it from cache
101  SDL_QueryTexture(cached_texture, nullptr, nullptr, &texW, &texH);
102  SDL_Rect Rect = {x, y, texW, texH};
103  SDL_SetTextureColorMod(cached_texture, color_font.r, color_font.g, color_font.b);
104  SDL_RenderCopy(sdl_renderer, cached_texture, nullptr, &Rect);
105  }else{ //else create a texture, render it, and then cache it - this is costly, so pick a good cache size
106  SDL_Surface* text = TTF_RenderUTF8_Blended(font, arg2.c_str(), color_font);
107  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, text);
108  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
109  SDL_Rect Rect = {x, y, texW, texH};
110  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
111  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
112  SDL_FreeSurface(text);
113  gFont_cache.cache_string(arg2, texture, arg, size);
114  }
115  }
116 }

130 void renderer_sdl::draw_text_cached_glyphs(const std::string arg, const std::string arg2, const int x, const int y,

135  std::vector<SDL_Texture*> tempVector = cached_vector->second;
136  if(!tempVector.empty()){
137  int texW, texH;

142  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);

144  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);

146  tempX += texW;
147  }

240 void renderer_sdl::cache_font(TTF_Font* Font, std::string font_and_size){
241  SDL_Color color_font = {255, 255, 255, 255};
242  char temp[2];
243  temp[1] = 0;
244  std::vector<SDL_Texture*> tempVector;
245  for(char j = 32; j < 127; j++){
246  temp[0] = j;
247  SDL_Surface* glyph = TTF_RenderUTF8_Blended(Font, temp, color_font);
248  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, glyph);
249  tempVector.push_back(texture);

251  }
252  fonts_cache[font_and_size] = tempVector;
