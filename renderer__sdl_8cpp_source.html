113 void renderer_sdl::draw_text_normal(std::string arg, std::string arg2, int x, int y, SDL_Color color_font , int size){
114  std::string font_and_size = arg+std::to_string(size);
115  TTF_Font* font = fonts[font_and_size];
116  if(font != nullptr){
117  int texW, texH;
118  if(gFont_cache.str_is_cached(arg2, arg, size)){ //if the given string (with same size and font) is already cached, get it from cache
119  SDL_Texture* texture = gFont_cache.return_cache();
120  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
121  SDL_Rect Rect = {x, y, texW, texH};
122  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
123  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
124  }else{ //else create a texture, render it, and then cache it - this is costly, so pick a good cache size
125  SDL_Surface* text = TTF_RenderUTF8_Blended(font, arg2.c_str(), color_font);
126  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, text);
127  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
128  SDL_Rect Rect = {x, y, texW, texH};
129  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
130  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
131  SDL_FreeSurface(text);
132  gFont_cache.cache_string(arg2, texture, arg, size);
133  }
134  }
135 }

149 void renderer_sdl::draw_text_cached(std::string arg, std::string arg2, int x, int y, SDL_Color color_font , int size){
150  std::string font_and_size = arg+std::to_string(size);
151  std::vector<SDL_Texture*> tempVector = fonts_cache[font_and_size];
152  if(!tempVector.empty()){
153  int texW, texH;
154  int tempX = x;
155  const char* arg3 = arg2.c_str();
156  for(int j = 0; arg3[j] != 0; j++){
157  SDL_Texture* texture = tempVector.at(static_cast<unsigned int>(arg3[j]-32));
158  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
159  SDL_Rect Rect = {tempX, y, texW, texH};
160  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
161  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
162  tempX += texW;
163  }
164  }

255 void renderer_sdl::cache_font(TTF_Font* Font, std::string font_and_size){
256  SDL_Color color_font = {255, 255, 255, 255};
257  char temp[2];
258  temp[1] = 0;
