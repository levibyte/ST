112 void ST::renderer_sdl::draw_text_lru_cached(const std::string& arg, const std::string& arg2, int x, int y, SDL_Color color_font, uint8_t size){
113  std::string font_and_size = arg+std::to_string(size);
114  TTF_Font* font = fonts[font_and_size];
115  if(font != nullptr){
116  int texW, texH;
117  SDL_Texture* cached_texture = font_cache::get_cached_string(arg2, arg, size);
118  if(cached_texture != nullptr){ //if the given string (with same size and font) is already cached, get it from cache
119  SDL_QueryTexture(cached_texture, nullptr, nullptr, &texW, &texH);
120  SDL_Rect Rect = {x, y, texW, texH};
121  SDL_SetTextureColorMod(cached_texture, color_font.r, color_font.g, color_font.b);
122  SDL_RenderCopy(sdl_renderer, cached_texture, nullptr, &Rect);
123  }else{ //else create a texture, render it, and then cache it - this is costly, so pick a good cache size
124  SDL_Surface* text = TTF_RenderUTF8_Blended(font, arg2.c_str(), color_font);
125  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, text);
126  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
127  SDL_Rect Rect = {x, y, texW, texH};
128  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
129  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
130  SDL_FreeSurface(text);
131  font_cache::cache_string(arg2, texture, arg, size);
132  }
133  }
134 }

148 void ST::renderer_sdl::draw_text_cached_glyphs(const std::string& arg, const std::string& arg2, const int x, const int y,
149  const SDL_Color color_font, const int size) {
150  std::string font_and_size = arg+std::to_string(size);
151  auto cached_vector = fonts_cache.find(font_and_size);
152  if(cached_vector != fonts_cache.end()){
153  std::vector<SDL_Texture*> tempVector = cached_vector->second;
154  if(!tempVector.empty()){
155  int texW, texH;
156  int tempX = x;
157  const char* arg3 = arg2.c_str();
158  for(int j = 0; arg3[j] != 0; j++){
159  SDL_Texture* texture = tempVector.at(static_cast<unsigned int>(arg3[j]-32));
160  SDL_QueryTexture(texture, nullptr, nullptr, &texW, &texH);
161  SDL_Rect Rect = {tempX, y, texW, texH};
162  SDL_SetTextureColorMod(texture, color_font.r, color_font.g, color_font.b);
163  SDL_RenderCopy(sdl_renderer, texture, nullptr, &Rect);
164  tempX += texW;
165  }
166  }
167  }

258 void ST::renderer_sdl::cache_font(TTF_Font* Font, const std::string& font_and_size){
259  SDL_Color color_font = {255, 255, 255, 255};
260  char temp[2];
261  temp[1] = 0;
262  std::vector<SDL_Texture*> tempVector;
263  for(char j = 32; j < 127; j++){
264  temp[0] = j;
265  SDL_Surface* glyph = TTF_RenderUTF8_Blended(Font, temp, color_font);
266  SDL_Texture* texture = SDL_CreateTextureFromSurface(sdl_renderer, glyph);
267  tempVector.push_back(texture);
268  SDL_FreeSurface(glyph);
269  }
270  fonts_cache[font_and_size] = tempVector;
271 }
