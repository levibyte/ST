176 void lua_backend::run_file(std::string file){

178  int status = luaL_loadbuffer(L, temp.c_str(), temp.size(), file.c_str());
179  if (status == LUA_ERRSYNTAX || status == LUA_ERRFILE || lua_pcall(L, 0, 0, 0)){
180  fprintf(stderr, "cannot compile script\n");
181  lua_error(L);
