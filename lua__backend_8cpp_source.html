175 void lua_backend::run_file(std::string file){

177  int status = luaL_loadbuffer(L, temp.c_str(), temp.size(), file.c_str());
178  if (status == LUA_ERRSYNTAX || status == LUA_ERRFILE || lua_pcall(L, 0, 0, 0)){
179  fprintf(stderr, "cannot compile script\n");
180  lua_error(L);
