158 void drawing_manager::process_lights(std::vector<ST::light>* lights){
159  #ifdef __DEBUG
160  if(!lighting_enabled){
161  return;
162  }
163  #endif
164  for(int i = 0; i < w_width; i++){
165  for(int j = 0; j < w_height; j++){
166  lightmap[i][j] = darkness_level;
167  }
168  }
169  for(unsigned int q = 0; q < lights->size(); q++){
170  int x = lights->at(q).origin_x - Camera.x;
171  int y = lights->at(q).origin_y - Camera.y;
172  double count = 0;
173  double step = darkness_level/(double)lights->at(q).radius;
174  count = 0;
175  int radius = lights->at(q).radius;
176  int intensity = lights->at(q).intensity;
177  if(x - radius - intensity > w_width || y - radius - intensity > w_height)
178  continue;
179  double step2 = 0;
180  for(int i = y; i < y + radius + intensity; i++){
181  for(int j = x; j < x + radius + intensity; j++){
182  if(j > 0 && j < w_width && i > 0 && i < w_height)
183  lightmap[j][i] = (uint8_t)lights->at(q).brightness + (uint8_t)count;
184  if(count + lights->at(q).brightness < darkness_level && j > x + intensity)
185  count += step;
186  }
187  count = 0;
188  count += step2;
189  if(step2 + lights->at(q).brightness < darkness_level && i > y + intensity)
190  step2 += step;
191  }
192  count = 0;
193  step2 = 0;
194  for(int i = y; i > y - radius - intensity; i--){
195  for(int j = x; j > x - radius - intensity; j--){
196  if(j > 0 && j < w_width && i > 0 && i < w_height)
197  lightmap[j][i] = lights->at(q).brightness + (uint8_t)count;
198  if(count + lights->at(q).brightness < darkness_level && j < x - intensity)
199  count += step;
200  }
201  count = 0;
202  count += step2;
203  if(step2 + lights->at(q).brightness < darkness_level && i < y - intensity)
204  step2 += step;
205  }
206  count = 0;
207  step2 = 0;
208  for(int i = y; i > y - radius - intensity; i--){
209  for(int j = x; j < x + radius + intensity; j++){
210  if(j > 0 && j < w_width && i > 0 && i < w_height)
211  lightmap[j][i] = lights->at(q).brightness + (uint8_t)count;
212  if(count + lights->at(q).brightness < darkness_level && j > x + intensity)
213  count += step;
214  }
215  count = 0;
216  count += step2;
217  if(step2 + lights->at(q).brightness < darkness_level && i < y - intensity)
218  step2 += step;
219  }
220  count = 0;
221  step2 = 0;
222  for(int i = y; i < y + radius + intensity; i++){
223  for(int j = x; j > x - radius - intensity; j--){
224  if(j > 0 && j < w_width && i > 0 && i < w_height)
225  lightmap[j][i] = lights->at(q).brightness + (uint8_t)count;
226  if(count + lights->at(q).brightness< darkness_level && j < x - intensity)
227  count += step;
228  }
229  count = 0;
230  count += step2;
231  if(step2 + lights->at(q).brightness < darkness_level && i > y + intensity) {
232  step2 += step;
233  }
234  }
235  }

241 void drawing_manager::draw_lights(){
242  #ifdef __DEBUG
243  if(!lighting_enabled){
244  return;
245  }
246  #endif
247  //Losing a lot of fps here :)
248  SDL_Rect tempRect = {0, 0, lights_quality, lights_quality};
249  for(int i = 0; i <= w_width-lights_quality; i += lights_quality){
250  int a = 1;
251  tempRect.x = i;
252  for(int j = 0; j <= w_height-lights_quality; j += lights_quality){
253  if(lightmap[i][j] == lightmap[i][j+lights_quality] && j+lights_quality == w_height){
254  tempRect.y = j - a*lights_quality;
255  tempRect.h = (a+1)*lights_quality;
256  SDL_Color light_color = {0, 0, 0, lightmap[i][j]};
257  gRenderer.draw_rectangle_filled(tempRect.x, tempRect.y, tempRect.w, tempRect.h, light_color);
258  }
259  else if(lightmap[i][j] == lightmap[i][j+lights_quality]){
260  a++;
261  }else{
262  tempRect.h = a*lights_quality;
263  tempRect.y = j - a*lights_quality;
264  SDL_Color light_color = {0, 0, 0, lightmap[i][j]};
265  gRenderer.draw_rectangle_filled(tempRect.x, tempRect.y, tempRect.w, tempRect.h, light_color);
266  a = 1;
267  }

277 void drawing_manager::handle_messages(){
278  message* temp = msg_sub.get_next_message();
279  while(temp != nullptr){
280  if(temp->msg_name == VSYNC_ON){
281  gRenderer.vsync_on();
282  gMessage_bus->send_msg(make_msg(VSYNC_IS_ON, nullptr));
283  }
284  else if(temp->msg_name == VSYNC_OFF){
285  gRenderer.vsync_off();
286  gMessage_bus->send_msg(make_msg(VSYNC_IS_OFF, nullptr));
287  }
288  else if(temp->msg_name == SET_DARKNESS){
289  set_darkness(*(uint8_t*)temp->get_data());
290  }
291  else if(temp->msg_name == SHOW_COLLISIONS){
292  auto arg = (bool*)temp->get_data();
293  if(*arg) {
294  show_collisions();
295  }
296  else if(!(*arg)){
297  hide_collisions();
298  }
299  }
300  else if(temp->msg_name == SHOW_FPS){
301  auto arg = (bool*)temp->get_data();
302  if(*arg) {
303  show_fps = true;
304  }
305  else if(!(*arg)){
306  show_fps = false;
307  }
308  }
309  else if(temp->msg_name == ENABLE_LIGHTING){
310  auto arg = (bool*)temp->get_data();
311  if(*arg) {
312  lighting_enabled = true;
313  }
314  else if(!(*arg)){
315  lighting_enabled = false;
316  }
317  }
318  else if(temp->msg_name == ASSETS) {
319  auto gAssets = (ST::assets **) temp->get_data();
320  ST::assets *temp_ast = *gAssets;
321  asset_ptr = temp_ast;
322  gRenderer.upload_surfaces(&asset_ptr->surfaces);
323  gRenderer.upload_fonts(&asset_ptr->fonts);
324  }
325  destroy_msg(temp);
326  temp = msg_sub.get_next_message();
327  }
328 }
