164 void drawing_manager::process_lights(const std::vector<ST::light>& lights){
165  #ifdef __DEBUG
166  if(!lighting_enabled){
167  return;
168  }
169  #endif
170  for(int i = 0; i < w_width; i++){
171  for(int j = 0; j < w_height; j++){
172  lightmap[i][j] = darkness_level;
173  }
174  }
175  for (const auto &light : lights) {
176  int x = light.origin_x - Camera.x;
177  int y = light.origin_y - Camera.y;
178  double count = 0;
179  double step = darkness_level/ static_cast<double>(light.radius);
180  count = 0;
181  int radius = light.radius;
182  int intensity = light.intensity;
183  if(x - radius - intensity > w_width || y - radius - intensity > w_height)
184  continue;
185  double step2 = 0;
186  for(int i = y; i < y + radius + intensity; i++){
187  for(int j = x; j < x + radius + intensity; j++){
188  if(j > 0 && j < w_width && i > 0 && i < w_height)
189  lightmap[j][i] = (uint8_t) light.brightness + (uint8_t)count;
190  if(count + light.brightness < darkness_level && j > x + intensity)
191  count += step;
192  }
193  count = 0;
194  count += step2;
195  if(step2 + light.brightness < darkness_level && i > y + intensity)
196  step2 += step;
197  }
198  count = 0;
199  step2 = 0;
200  for(int i = y; i > y - radius - intensity; i--){
201  for(int j = x; j > x - radius - intensity; j--){
202  if(j > 0 && j < w_width && i > 0 && i < w_height)
203  lightmap[j][i] = light.brightness + (uint8_t)count;
204  if(count + light.brightness < darkness_level && j < x - intensity)
205  count += step;
206  }
207  count = 0;
208  count += step2;
209  if(step2 + light.brightness < darkness_level && i < y - intensity)
210  step2 += step;
211  }
212  count = 0;
213  step2 = 0;
214  for(int i = y; i > y - radius - intensity; i--){
215  for(int j = x; j < x + radius + intensity; j++){
216  if(j > 0 && j < w_width && i > 0 && i < w_height)
217  lightmap[j][i] = light.brightness + (uint8_t)count;
218  if(count + light.brightness < darkness_level && j > x + intensity)
219  count += step;
220  }
221  count = 0;
222  count += step2;
223  if(step2 + light.brightness < darkness_level && i < y - intensity)
224  step2 += step;
225  }
226  count = 0;
227  step2 = 0;
228  for(int i = y; i < y + radius + intensity; i++){
229  for(int j = x; j > x - radius - intensity; j--){
230  if(j > 0 && j < w_width && i > 0 && i < w_height)
231  lightmap[j][i] = light.brightness + (uint8_t)count;
232  if(count + light.brightness< darkness_level && j < x - intensity)
233  count += step;
234  }
235  count = 0;
236  count += step2;
237  if(step2 + light.brightness < darkness_level && i > y + intensity) {
238  step2 += step;
239  }
240  }
241  }
242 }

247 void drawing_manager::draw_lights() const{
248  //Losing a lot of fps here :)
249  SDL_Rect tempRect = {0, 0, lights_quality, lights_quality};
250  for(int i = 0; i <= w_width-lights_quality; i += lights_quality){
251  int a = 1;
252  tempRect.x = i;
253  for(int j = 0; j <= w_height-lights_quality; j += lights_quality){
254  if(lightmap[i][j] == lightmap[i][j+lights_quality] && j+lights_quality == w_height){
255  tempRect.y = j - a*lights_quality;
256  tempRect.h = (a+1)*lights_quality;
257  SDL_Color light_color = {0, 0, 0, lightmap[i][j]};
258  gRenderer.draw_rectangle_filled(tempRect.x, tempRect.y, tempRect.w, tempRect.h, light_color);
259  }
260  else if(lightmap[i][j] == lightmap[i][j+lights_quality]){
261  a++;
262  }else{
263  tempRect.h = a*lights_quality;
264  tempRect.y = j - a*lights_quality;
265  SDL_Color light_color = {0, 0, 0, lightmap[i][j]};
266  gRenderer.draw_rectangle_filled(tempRect.x, tempRect.y, tempRect.w, tempRect.h, light_color);
267  a = 1;
268  }

278 void drawing_manager::handle_messages(){
279  message* temp = msg_sub.get_next_message();
280  while(temp != nullptr){
281  if(temp->msg_name == VSYNC_ON){
282  gRenderer.vsync_on();
283  gMessage_bus->send_msg(make_msg(VSYNC_IS_ON, nullptr));
284  }
285  else if(temp->msg_name == VSYNC_OFF){
286  gRenderer.vsync_off();
287  gMessage_bus->send_msg(make_msg(VSYNC_IS_OFF, nullptr));
288  }
289  else if(temp->msg_name == SET_DARKNESS){
290  set_darkness(*(uint8_t*)temp->get_data());
291  }
292  else if(temp->msg_name == SHOW_COLLISIONS){
293  auto arg = static_cast<bool*>(temp->get_data());
294  if(*arg) {
295  show_collisions();
296  }
297  else if(!(*arg)){
298  hide_collisions();
299  }
300  }
301  else if(temp->msg_name == SHOW_FPS){
302  auto arg = static_cast<bool*>(temp->get_data());
303  if(*arg) {
304  show_fps = true;
305  }
306  else if(!(*arg)){
307  show_fps = false;
308  }
309  }
310  else if(temp->msg_name == ENABLE_LIGHTING){
311  auto arg = static_cast<bool*>(temp->get_data());
312  if(*arg) {
313  lighting_enabled = true;
314  }
315  else if(!(*arg)){
316  lighting_enabled = false;
317  }
318  }
319  else if(temp->msg_name == ASSETS) {
320  auto gAssets = static_cast<ST::assets**>(temp->get_data());
321  ST::assets *temp_ast = *gAssets;
322  asset_ptr = temp_ast;
323  gRenderer.upload_surfaces(&asset_ptr->surfaces);
324  gRenderer.upload_fonts(&asset_ptr->fonts);
325  }
326  destroy_msg(temp);
327  temp = msg_sub.get_next_message();
328  }
329 }
