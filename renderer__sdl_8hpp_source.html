83 inline void renderer_sdl::draw_lights(uint8_t arg[1920][1080]){
84  int k = 0;
85  for(uint16_t i = 0; i < height; i++){
86  for(uint16_t j = 0; j < width; j++){
87  pixels[k] = arg[j][i] << (uint8_t)24;
88  k++;
89  };
90  }
91  SDL_UpdateTexture(lights_texture, nullptr, pixels, width*sizeof(Uint32));
92  SDL_RenderCopy(sdl_renderer, lights_texture, nullptr, nullptr);
93 }

162 inline void renderer_sdl::draw_sprite(size_t arg, int x, int y, int sprite, int animation, int animation_num, int sprite_num){
163  SDL_Texture* temp = textures[arg];
164  if(temp != nullptr){
165  int tex_w, tex_h;
166  SDL_QueryTexture(temp, nullptr, nullptr, &tex_w, &tex_h);
167  int temp1 = tex_h/animation_num;
168  int temp2 = tex_w/sprite_num;
169  SDL_Rect src_rect = {x, y - temp1, temp2, temp1};
170  SDL_Rect dst_rect = {sprite*(tex_w/sprite_num), temp1*(animation-1), temp2, temp1};
171  SDL_RenderCopy(sdl_renderer, temp, &dst_rect, &src_rect);
172  }
173 }
