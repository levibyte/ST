74 inline void renderer_sdl::draw_lights(uint8_t arg[1920][1080]){
75  int k = 0;
76  for(uint16_t i = 0; i < height; i++){
77  for(uint16_t j = 0; j < width; j++){
78  pixels[k] = arg[j][i] << (uint8_t)24;
79  k++;
80  };
81  }
82  SDL_UpdateTexture(lights_texture, nullptr, pixels, width*sizeof(Uint32));
83  SDL_RenderCopy(sdl_renderer, lights_texture, nullptr, nullptr);
84 }

117 inline void renderer_sdl::draw_sprite(size_t arg, int x, int y, int sprite, int animation, int animation_num, int sprite_num){
118  SDL_Texture* temp = textures[arg];
119  if(temp != nullptr){
120  int tex_w, tex_h;
121  SDL_QueryTexture(temp, nullptr, nullptr, &tex_w, &tex_h);
122  int temp1 = tex_h/animation_num;
123  int temp2 = tex_w/sprite_num;
124  SDL_Rect src_rect = {x, y - temp1, temp2, temp1};
125  SDL_Rect dst_rect = {sprite*(tex_w/sprite_num), temp1*(animation-1), temp2, temp1};
126  SDL_RenderCopy(sdl_renderer, temp, &dst_rect, &src_rect);
127  }
128 }
