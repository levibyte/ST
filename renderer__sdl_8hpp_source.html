84 inline void renderer_sdl::draw_texture(const size_t arg, int x, int y) const{
85  auto texture = textures.find(arg);
86  if (texture != textures.end()) {
87  int tex_w, tex_h;
88  SDL_QueryTexture(texture->second, nullptr, nullptr, &tex_w, &tex_h);
89  SDL_Rect src_rect = {x, y - tex_h, tex_w, tex_h};

145 inline void renderer_sdl::draw_sprite(size_t arg, int x, int y, int sprite, int animation, int animation_num, int sprite_num) const{
146  auto texture = textures.find(arg);
147  if(texture != textures.end()){
148  int tex_w, tex_h;
149  SDL_QueryTexture(texture->second, nullptr, nullptr, &tex_w, &tex_h);
150  int temp1 = tex_h/animation_num;
151  int temp2 = tex_w/sprite_num;
152  SDL_Rect dst_rect = {x, y - temp1, temp2, temp1};
153  SDL_Rect src_rect = {sprite*(tex_w/sprite_num), temp1*(animation-1), temp2, temp1};
154  SDL_RenderCopy(sdl_renderer, texture->second, &src_rect, &dst_rect);
155  }
156 }
