58 inline ST::task* task_allocator::allocate_task(void (*function)(void *), void *arg, SDL_semaphore *dependency, int affinity){
59  uint16_t i = 0;
60  SDL_LockMutex(access_mutex);
61  //find the next free spot in memory
62  while(allocated[pointer] && i < memory_size){
63  pointer++;
64  i++;
65  if(pointer > memory_size-1){
66  pointer = 0;
67  }
68  }
69  if(i == memory_size){
70  return nullptr;
71  }
72  allocated[pointer] = true;
73  auto temp = new (memory+pointer) ST::task(pointer, function, arg, dependency, affinity);
74  SDL_UnlockMutex(access_mutex);
75  return temp;
76 }
