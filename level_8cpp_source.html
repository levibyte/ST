80 int ST::level::load_input_conf(){
81  std::ifstream file;
82  std::string temp = "levels/" + name;
83  temp = temp + "/inputConf.cfg";
84  file.open(temp.c_str());
85  if(file.is_open()){
86  log(INFO, "Loading " + temp);
87  std::string action;
88  std::string button;
89  int actionRead = 0;
90  while(!file.eof()){
91  auto a = static_cast<char>(file.get());
92 
93  //Ignore comments
94  if(a == '#'){
95  while(a != '\n'){
96  a = static_cast<char>(file.get());;
97  }
98  }
99  else if(a == ' ') {
100  continue;
101  }
102  else if(a == '=') {
103  actionRead = 1;
104  }
105  else if(actionRead == 0 && a == '\n'){ //ignore empty lines
106  continue;
107  }
108  else if(actionRead == 0)
109  action += a;
110  else if(a == '\n' || file.eof()){
111  std::hash<std::string> hash_f;
112  data.actions_Buttons[hash_f(action)] = key_index(button);
113  actionRead = 0;
114  action.clear();
115  button.clear();
116  }
117  else{
118  button += a;
119  }
120  }
121  file.close();
122  }
123  else{
124  log(ERROR, "File " + temp + " not found");
125  return -1;
126  }
127  return 0;
128 }
