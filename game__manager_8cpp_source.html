67 void game_manager::handle_messages(){
68  message* temp = msg_sub.get_next_message();
69  while(temp != nullptr){
70  if(temp->msg_name == LOAD_LEVEL){
71  auto name = static_cast<std::string*>(temp->get_data());
72  load_level(*name);
73  }
74  else if(temp->msg_name == START_LEVEL){
75  auto name = static_cast<std::string*>(temp->get_data());
76  start_level(*name);
77  }
78  else if(temp->msg_name == UNLOAD_LEVEL){
79  auto name = static_cast<std::string*>(temp->get_data());
80  unload_level(*name);
81  }
82  else if(temp->msg_name == KEY_PRESSED){
83  uint8_t key_index = *static_cast<uint8_t*>(temp->get_data());
84  keys_pressed_data[key_index] = true;
85  keys_held_data[key_index] = false;
86  keys_released_data[key_index] = false;
87  }
88  else if(temp->msg_name == KEY_HELD){
89  uint8_t key_index = *static_cast<uint8_t*>(temp->get_data());
90  keys_pressed_data[key_index] = false;
91  keys_held_data[key_index] = true;
92  keys_released_data[key_index] = false;
93  }
94  else if(temp->msg_name == KEY_RELEASED){
95  uint8_t key_index = *static_cast<uint8_t*>(temp->get_data());
96  keys_pressed_data[key_index] = false;
97  keys_held_data[key_index] = false;
98  keys_released_data[key_index] = true;
99  }
100  else if(temp->msg_name == MOUSE_X){
101  int32_t x = *static_cast<int32_t*>(temp->get_data());
102  mouse_x = x;
103  }
104  else if(temp->msg_name == MOUSE_Y){
105  int32_t y = *static_cast<int32_t*>(temp->get_data());
106  mouse_y = y;
107  }
108  else if(temp->msg_name == VOLUME_LEVEL){
109  uint8_t val = *static_cast<uint8_t*>(temp->get_data());
110  volume_level = val;
111  }
112  else if(temp->msg_name == VSYNC_IS_ON){
113  vsync_flag = true;
114  }
115  else if(temp->msg_name == VSYNC_IS_OFF){
116  vsync_flag = false;
117  }
118  else if(temp->msg_name == END_GAME){
119  SDL_AtomicSet(&end_game, 0);
120  }
121  else if(temp->msg_name == SHOW_MOUSE){
122  auto arg = static_cast<bool*>(temp->get_data());
123  if(*arg){
124  SDL_ShowCursor(SDL_ENABLE);
125  }else{
126  SDL_ShowCursor(SDL_DISABLE);
127  }
128  }
129  else if(temp->msg_name == EXECUTE_SCRIPT){
130  auto scipt = static_cast<std::string*>(temp->get_data());
131  gScript_backend.run_script(*scipt);
132  }
133  else if(temp->msg_name == FULLSCREEN_STATUS){
134  bool arg = *static_cast<bool*>(temp->get_data());
135  fullscreen_status = arg;
136  }
137  destroy_msg(temp);
138  temp = msg_sub.get_next_message();
139  }
140 }

146 void game_manager::load_level(const std::string& level_name){
147 
148  //check if it is already loaded.
149  for(auto& i: levels) {
150  if (i.get_name() == level_name) {
151  return;
152  }
153  }
154 
155  //otherwise - create it
156  auto temp = ST::level(level_name, gMessage_bus);
157  temp.load();
158  levels.push_back(temp);
159 
160  //current level pointer must be reset, because apparently adding to the vector changes the pointer address
161  //(makes sense as it has to reallocate the whole thing)
162  for (auto &level : levels) {
163  if(level.get_name() == active_level){
164  current_level_pointer = &level;
165  break;
166  }
167  }
168 }

187 void game_manager::start_level(const std::string& level_name){
188  gScript_backend.close();

190  active_level = level_name;
191 
192  //set current level pointer

197  }
198  }
199  get_level_data()->lights.clear();

201

205 
206  std::string temp = "levels/";

208  temp = temp + "/level.lua";
209  gScript_backend.run_file(temp);
210 
211  //load level loop
212  temp = "levels/";
213  temp = temp + active_level;
214  temp = temp + "/loop.lua";
215  gScript_backend.load_file(temp);
216  gScript_backend.set_global("loop");
217 
218  //Register all the keys this level uses with the input manager.
219  for(auto i : get_level_data()->actions_Buttons) {
220  gMessage_bus->send_msg(make_msg(REGISTER_KEY, make_data(i.second)));
221  }
