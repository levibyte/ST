70 void window_manager::handle_messages(){
71  message* temp = msg_sub.get_next_message();
72  while(temp != nullptr){
73  if(temp->msg_name == SET_FULLSCREEN){
74  auto arg = static_cast<bool*>(temp->get_data());
75  set_fullscreen(*arg);
76  gMessage_bus->send_msg(make_msg(FULLSCREEN_STATUS, make_data<bool>(*arg)));
77  }
78  else if(temp->msg_name == SET_WINDOW_BRIGHTNESS){
79  auto arg = static_cast<float*>(temp->get_data());
80  set_brightness(*arg);
81  gMessage_bus->send_msg(make_msg(LOG_SUCCESS, make_data<std::string>("Brightness set to: " + std::to_string(*arg))));
82  }
83  else if(temp->msg_name == SET_WINDOW_RESOLUTION){
84  auto res = *static_cast<std::tuple<int16_t, int16_t>*>(temp->get_data());
85  gMessage_bus->send_msg(make_msg(REAL_SCREEN_COORDINATES, make_data(res)));
86  SDL_SetWindowSize(window, std::get<0>(res), std::get<1>(res));
87  SDL_SetWindowPosition(window, SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED);
88  width = std::get<0>(res);
89  height = std::get<1>(res);
90  }
91  destroy_msg(temp);
92  temp = msg_sub.get_next_message();
93  }
94 }
