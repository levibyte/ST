43 void console::handle_messages(){
44  message* temp = msg_sub.get_next_message();
45  while(temp != nullptr){
46  if(temp->msg_name == CONSOLE_WRITE){
47  auto log = static_cast<ST::console_log*>(temp->get_data());
48  if(log_level == 0x01) {
49  if (log->type == ST::log_type::ERROR) {
50  write(*log);
51  }
52  }else if(log_level == 0x02){
53  if(log->type == ST::log_type::SUCCESS){
54  write(*log);
55  }
56  }else if(log_level == 0x04){
57  if(log->type == ST::log_type::INFO){
58  write(*log);
59  }
60  }else if(log_level == 0x03){
61  if(log->type == ST::log_type::ERROR || log->type == ST::log_type::SUCCESS){
62  write(*log);
63  }
64  }else if(log_level == 0x05){
65  if(log->type == ST::log_type::ERROR || log->type == ST::log_type::INFO){
66  write(*log);
67  }
68  }else if(log_level == 0x06){
69  if(log->type == ST::log_type::SUCCESS || log->type == ST::log_type::INFO){
70  write(*log);
71  }
72  }else if(log_level == 0x07){
73  write(*log);
74  }
75  }
76  else if(temp->msg_name == CONSOLE_TOGGLE){
77  toggle();
78  }
79  else if(temp->msg_name == MOUSE_SCROLL){
80  int value = *static_cast<int*>(temp->get_data());
81  scroll(value);
82  }
83  else if(temp->msg_name == KEY_PRESSED){
84  auto key_val = static_cast<ST::key*>(temp->get_data());
85  if(*key_val == ST::key::ENTER){
86  if(!composition.empty()){
87  write(ST::console_log(ST::log_type::INFO, composition));
88  gMessage_bus->send_msg(make_msg(EXECUTE_SCRIPT, make_data(composition)));
89  }
90  composition.clear();
91  gMessage_bus->send_msg(make_msg(CLEAR_TEXT_STREAM, nullptr));
92  }
93  }
94  else if(temp->msg_name == TEXT_STREAM){
95  composition = *static_cast<std::string*>(temp->get_data());
96  }
97  destroy_msg(temp);
98  temp = msg_sub.get_next_message();
99  }
100 }
