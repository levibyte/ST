44 void console::handle_messages(){
45  message* temp = msg_sub.get_next_message();
46  while(temp != nullptr){
47  if(temp->msg_name == CONSOLE_WRITE){
48  auto log = static_cast<ST::console_log*>(temp->get_data());
49  if(log_level == 0x01) {
50  if (log->type == ST::log_type::ERROR) {
51  write(*log);
52  }
53  }else if(log_level == 0x02){
54  if(log->type == ST::log_type::SUCCESS){
55  write(*log);
56  }
57  }else if(log_level == 0x04){
58  if(log->type == ST::log_type::INFO){
59  write(*log);
60  }
61  }else if(log_level == 0x03){
62  if(log->type == ST::log_type::ERROR || log->type == ST::log_type::SUCCESS){
63  write(*log);
64  }
65  }else if(log_level == 0x05){
66  if(log->type == ST::log_type::ERROR || log->type == ST::log_type::INFO){
67  write(*log);
68  }
69  }else if(log_level == 0x06){
70  if(log->type == ST::log_type::SUCCESS || log->type == ST::log_type::INFO){
71  write(*log);
72  }
73  }else if(log_level == 0x07){
74  write(*log);
75  }
76  }
77  else if(temp->msg_name == CONSOLE_TOGGLE){
78  toggle();
79  }
80  else if(temp->msg_name == CONSOLE_CLEAR){
81  this->entries.clear();
82  }
83  else if(temp->msg_name == MOUSE_SCROLL){
84  int value = *static_cast<int*>(temp->get_data());
85  scroll(value);
86  }
87  else if(temp->msg_name == KEY_PRESSED){
88  auto key_val = static_cast<ST::key*>(temp->get_data());
89  if(*key_val == ST::key::ENTER){
90  if(!composition.empty()){
91  write(ST::console_log(ST::log_type::INFO, composition));
92  gMessage_bus->send_msg(make_msg(EXECUTE_SCRIPT, make_data(composition)));
93  }
94  composition.clear();
95  gMessage_bus->send_msg(make_msg(CLEAR_TEXT_STREAM, nullptr));
96  }
97  }
98  else if(temp->msg_name == TEXT_STREAM){
99  composition = *static_cast<std::string*>(temp->get_data());
100  }
101  destroy_msg(temp);
102  temp = msg_sub.get_next_message();
103  }
104 }
