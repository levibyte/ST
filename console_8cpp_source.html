45 void console::handle_messages(){
46  message* temp = msg_sub.get_next_message();
47  while(temp != nullptr){
48  if(temp->msg_name == CONSOLE_WRITE){
49  auto log = (ST::console_log*)temp->get_data();
50  if(log_level == 0x01) {
51  if (log->type == ST::log_type::ERROR) {
52  write(*log);
53  }
54  }else if(log_level == 0x02){
55  if(log->type == ST::log_type::SUCCESS){
56  write(*log);
57  }
58  }else if(log_level == 0x04){
59  if(log->type == ST::log_type::INFO){
60  write(*log);
61  }
62  }else if(log_level == 0x03){
63  if(log->type == ST::log_type::ERROR || log->type == ST::log_type::SUCCESS){
64  write(*log);
65  }
66  }else if(log_level == 0x05){
67  if(log->type == ST::log_type::ERROR || log->type == ST::log_type::INFO){
68  write(*log);
69  }
70  }else if(log_level == 0x06){
71  if(log->type == ST::log_type::SUCCESS || log->type == ST::log_type::INFO){
72  write(*log);
73  }
74  }else if(log_level == 0x07){
75  write(*log);
76  }
77  }
78  else if(temp->msg_name == CONSOLE_TOGGLE){
79  toggle();
80  }
81  else if(temp->msg_name == MOUSE_SCROLL){
82  int value = *(int*)temp->get_data();
83  scroll(value);
84  }
85  else if(temp->msg_name == KEY_PRESSED){
86  auto key_val = (ST::key*)temp->get_data();
87  if(*key_val == ST::key::ENTER){
88  write(ST::console_log(ST::log_type ::INFO, composition));
89  if(!composition.empty()){
90  gMessage_bus->send_msg(make_msg(EXECUTE_SCRIPT, make_data<std::string>(composition)));
91  }
92  gMessage_bus->send_msg(make_msg(CLEAR_TEXT_STREAM, nullptr));
93  }
94  }
95  else if(temp->msg_name == TEXT_STREAM){
96  composition = *(std::string*)temp->get_data();
97  }
98  destroy_msg(temp);
99  temp = msg_sub.get_next_message();
100  }
101 }
