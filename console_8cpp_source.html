49 void console::handle_messages(){
50  message* temp = msg_sub.get_next_message();
51  while(temp != nullptr){
52  if(temp->msg_name == LOG_ERROR){
53  auto log = static_cast<std::string*>(temp->get_data());
54  if(log_level == 0x01 || log_level >= 0x03) {
55  write(*log, ST::log_type::ERROR);
56  }
57  }else if(temp->msg_name == LOG_INFO){
58  auto log = static_cast<std::string*>(temp->get_data());
59  if(log_level == 0x04 || log_level >= 0x05) {
60  write(*log, ST::log_type::INFO);
61  }
62  }else if(temp->msg_name == LOG_SUCCESS){
63  auto log = static_cast<std::string*>(temp->get_data());
64  if(log_level == 0x02 || log_level == 0x03 || log_level > 0x06) {
65  write(*log, ST::log_type::SUCCESS);
66  }
67  }
68  else if(temp->msg_name == CONSOLE_TOGGLE){
69  toggle();
70  }
71  else if(temp->msg_name == CONSOLE_CLEAR){
72  this->entries.clear();
73  }
74  else if(temp->msg_name == MOUSE_SCROLL){
75  int value = *static_cast<int32_t*>(temp->get_data());
76  scroll(value);
77  }
78  else if(temp->msg_name == KEY_PRESSED){
79  auto key_val = static_cast<ST::key*>(temp->get_data());
80  if(*key_val == ST::key::ENTER){
81  if(!composition.empty()){
82  write(composition, ST::log_type ::INFO);
83  gMessage_bus->send_msg(make_msg(EXECUTE_SCRIPT, make_data(composition)));
84  }
85  composition.clear();
86  gMessage_bus->send_msg(make_msg(CLEAR_TEXT_STREAM, nullptr));
87  }else if(*key_val == ST::key::TILDE) {
88  toggle();
89  }
90  }
91  else if(temp->msg_name == TEXT_STREAM){
92  composition = *static_cast<std::string*>(temp->get_data());
93  }
94  destroy_msg(temp);
95  temp = msg_sub.get_next_message();
96  }
97 }
